\begin{table}[t]
\centering
\IfFileExists{cv5_injection_table.tex}{\input{cv5_injection_table.tex}}{\emph{(injection-site ablation table renders here as runs land)}}
\caption{\textbf{Where should the biological interaction enter? (main contribution).}
For every dataset we inject the interaction graph at each site in isolation and together,
under the same 5-fold CV: \emph{Embedding only} (the graph warm-starts and smooths the
learned token embedding), \emph{Router only} (the graph's centrality is a prior on the
depth router), and \emph{Both}~($=$~bioMoR). For single-cell the graph is the co-expression
interaction matrix; for multi-omics it is the \emph{provided} Reactome pathway$\to$pathway
graph (\texttt{adjacency\_matrix.csv}), never a co-expression matrix computed on the sparse
mutation/CNV data. Injecting at the embedding site dominates the router site, and the two
are complementary---\textbf{Both} is best, defining bioMoR.}
\label{tab:injection}
\end{table}

\caption{\textbf{Where biology should enter the router.} We ablate the depth router under
the same 5-fold CV on Baron (single-cell) and Prostate (multi-omics), comparing a fixed
\emph{biology-only} graph, a \emph{data-only} learned graph, and their combination
(\emph{biology $+$ data}, bioMoR). The data-only learned graph already matches or beats the
fixed biology-only graph, and combining the two is best---biological structure helps only
when it is learned jointly with the data, not imposed.}
\label{fig:biorouter}
\end{figure}
